import java.io.*;

public class Main {
	static void swap(int[] arr, int i, int j) {
		int tmp = arr[i];
		arr[i] = arr[j];
		arr[j] = tmp;
	}
	
	static boolean nextPermutation(int[] a, int n) {
		int i = n-1;
		while(i > 0 && a[i-1] > a[i]) i--;
		if(i <= 0) return false;
		
		int j = n-1;
		while(a[j] <= a[i-1]) j--;
		swap(a, i-1, j);
		
		j = n-1;
		while(i < j) {
			swap(a, i, j);
			i++; j--;
		}
		return true;
	}
	
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		int N = Integer.parseInt(br.readLine());
		int[] A = new int[N];
		for(int i=0; i < N; i++)
			A[i] = i + 1;
		
		for(int i=0; i < N; i++)               // 2번 풀이처럼 do~while 구문을 쓰면 두 번 나눠쓰지 않아도 됨.
			System.out.print(A[i] + " ");
		System.out.println();
		
		while(nextPermutation(A, N)) {
			for(int i=0; i < N; i++)
				System.out.print(A[i] + " ");
			System.out.println();
		}
		
		br.close();
	}
}

import java.io.*;

public class Main {
	static void swap(int[] arr, int i, int j) {
		int tmp = arr[i];
		arr[i] = arr[j];
		arr[j] = tmp;
	}
	
	static boolean nextPermutation(int[] a, int n) {
		int i = n-1;
		while(i > 0 && a[i-1] > a[i]) i--;
		if(i <= 0) return false;
		
		int j = n-1;
		while(a[j] <= a[i-1]) j--;
		swap(a, i-1, j);
		
		j = n-1;
		while(i < j) {
			swap(a, i, j);
			i++; j--;
		}
		return true;
	}
	
	public static void main(String[] args) throws IOException {
		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		int N = Integer.parseInt(br.readLine());
		int[] A = new int[N];
		for(int i=0; i < N; i++)
			A[i] = i + 1;
		
		do {
			for(int i=0; i < N; i++)
				System.out.print(A[i] + " ");
			System.out.println();
		} while(nextPermutation(A, N));
		
		br.close();
	}
}
